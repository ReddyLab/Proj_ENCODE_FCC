dat = dat_lmpra_arrange

    dplyr::mutate(
        Category = recode(
            Category, 
            "potential enhancer" = "potential\nenhancer",
            "negative control, shuffled" = "negative control\n(shuffled)",
            "negative control (Ernst et al 2016)" = "negative control\n(Ernst et al 2016)",
            "positive control (Ernst et al 2016)" = "positive control\n(Ernst et al 2016)"
        ))

### create plot
gpt = ggplot(dat, aes(x=Score, group=Category, fill=Category)) +
    geom_density(alpha=0.5) +

        title="Lenti-MPRA Log2FC", 
        x="Log2FC", 
        y="Density") +

Z Score transformation

Code 
dat = dat_lmpra_score_negative_shuffled
vec = dat$Score
val_mu = mean(vec)
val_sd = sd(vec)

dat = dat_lmpra_arrange
dat = dat %>% dplyr::mutate(ZScore = (Score - val_mu) / val_sd)

Visualization

Code 
### arrange data for plotting
dat = dat_lmpra_zscore

vec = c("potential enhancer", "promoter", "intergenic (7 loci)")
dat = dat %>% dplyr::filter(Category %in% vec)

dat = dat %>% dplyr::select(Score, ZScore)
colnames(dat) = c("Log2FC", "ZScore")
dat = dat %>% tidyr::gather(Group, Score)

### create plot
gpt = ggplot(dat, aes(x=Score, fill=Group)) + 
    geom_density(alpha=0.5) + 
    geom_vline(xintercept = 0, color = "grey30", linewidth=1) +
    labs(
        y="Density",
        title="Lenti-MPRA Log2FC vs Z-Score\n(Promoters/Enhancers/Intergenic regions)") +
    theme_cowplot() +
    background_grid() +
    theme_text
