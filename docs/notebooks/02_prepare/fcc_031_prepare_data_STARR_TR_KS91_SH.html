for idx in "${!FN_INPS[@]}"
do
    FN_INP=${FN_INPS[idx]}
    FN_OUT=${FN_OUTS[idx]}
    
    FILE=${FD_INP}/${FN_INP}
    LINK=${FD_OUT}/${FN_OUT}
    
    if [[ -h ${LINK} ]]; then
        echo "Remove existing link."
        rm ${LINK}
    fi
    
    if [[ ! -e ${LINK} ]]; then
        echo "Create symbolic links."
        ln -s ${FILE} ${LINK}
    fi
done
