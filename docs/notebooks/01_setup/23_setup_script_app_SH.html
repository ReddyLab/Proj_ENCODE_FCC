Set template

Code 
FDIRY=${FD_EXE}/app_duke_hardac
FNAME=sing_template.sh
FPATH=${FDIRY}/${FNAME}

cat > ${FPATH} << 'EOF'
#!/bin/bash

#########################################
### Wrapper of PLACEHOLDER_CMD container
### =====================================

IMAGE=PLACEHOLDER_IMG

CMD="PLACEHOLDER_CMD"
ARGS="$@"

singularity exec   \
    -B ${PWD}      \
    -B /data:/data \
    -B /gpfs:/gpfs \
    ${IMAGE} ${CMD} ${ARGS}
EOF

cat ${FPATH}
#!/bin/bash

#########################################
### Wrapper of PLACEHOLDER_CMD container
### =====================================

IMAGE=PLACEHOLDER_IMG

CMD="PLACEHOLDER_CMD"
ARGS="$@"

singularity exec   \
    -B ${PWD}      \
    -B /data:/data \
    -B /gpfs:/gpfs \
    ${IMAGE} ${CMD} ${ARGS}
Code 
CMD="bedtools"
IMG=${FD_SING}/biocontainer/"bedtools:2.31.1--hf5e1c6e_1"

FD_INP=${FD_EXE}/app_duke_hardac
FN_INP=sing_template.sh
FP_INP=${FD_INP}/${FN_INP}

FD_OUT=${FD_INP}
FN_OUT=${CMD}
FP_OUT=${FD_OUT}/${FN_OUT}

cp ${FP_INP} ${FP_OUT}
sed -i "s|PLACEHOLDER_IMG|${IMG}|g" ${FP_OUT}
sed -i "s|PLACEHOLDER_CMD|${CMD}|g" ${FP_OUT}
chmod +x ${FP_OUT}

cat ${FP_OUT}
