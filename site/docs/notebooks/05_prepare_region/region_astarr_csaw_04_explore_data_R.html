### set text size
theme_text = theme(
    title      = element_text(size = 16),
    axis.title = element_text(size = 16),
    axis.text  = element_text(size = 14)
)

### label and set factor
dat = dat_region_arrange
vec = c("KSMerge (6Dna7Rna)", "KS91 (6Dna4Rna)")
dat = dat %>% dplyr::mutate(Label = factor(Label, levels = vec))

### summarize the number of regions (total)
tmp = dat
vec = tmp %>%
    group_by(Label) %>%
    summarise(Count = n(), .groups="drop") %>%
    dplyr::mutate(
        line1 = Label,
        line2 = paste("Count =", scales::comma(Count)),
        combined = paste0(line1, "\n", line2)
    ) %>%
    { set_names(.$combined, .$Label) }
vec_txt_label = vec

### plot the score distribution (total)
gpt = ggplot(tmp, aes(x=Log2FC, color=Label)) +
    geom_density(linewidth = 1, alpha = 0.5) +
    scale_color_discrete(
        ### replace each original level by our multiâ€line version
        labels = vec_txt_label
    ) +
    labs(x="Log2(fold-change)", y="Density", title="ASTARR CSAW Regions (Total)")
gpt1 = gpt


### summarize the number of regions (significant)
tmp = dat %>% dplyr::filter(Signif3)
vec = tmp %>%
    group_by(Label) %>%
    summarise(Count = n(), .groups="drop") %>%
    dplyr::mutate(
        line1 = Label,
        line2 = paste("Count =", scales::comma(Count)),
        combined = paste0(line1, "\n", line2)
    ) %>%
    { set_names(.$combined, .$Label) }
vec_txt_label = vec

### plot the score distribution (significant)
gpt = ggplot(tmp, aes(x=Log2FC, color=Label)) +
    geom_density(linewidth = 1, alpha = 0.5) +
    scale_color_discrete(
        ### replace each original level by our multiâ€line version
        labels = vec_txt_label
    ) +
    labs(x="Log2(fold-change)", y="Density", title="ASTARR CSAW Regions (q-value < 0.001)")
gpt2 = gpt

options(repr.plot.height=4, repr.plot.width=10)
lst = list(gpt1, gpt2)
lst = lapply(lst, function(gpt){
    gpt = gpt +
        theme_cowplot() +
        background_grid() +
        theme_text +
        theme(
            legend.position = "inside", 
            legend.position.inside = c(0.1, 0.6),
            legend.background = element_rect(fill = "white"),
            legend.key.spacing.y = unit(0.5, 'cm')
        )
    return(gpt)
})

plt = plot_grid(plotlist = lst, ncol = 1, align = "v")
plt_export = plt

options(repr.plot.height=6, repr.plot.width=6)
