lst = lapply(VEC_TXT_FOLDER, function(txt_folder){
    ### get talbes
    dat_region_fcc = lst_dat_region_annot_fcc_starrmpra_peak_import[[txt_folder]]
    dat_region_nuc = lst_dat_region_nuc_import[[txt_folder]]

    ### get the fcc columns
    dat = dat_region_fcc
    dat = dat %>% 
        dplyr::select(Region, Group, Label, Direction_Assay) %>%
        dplyr::distinct()
    dat_region_fcc_arrange = dat
    
    ### join the tables
    dat = dat_region_fcc_arrange
    dat = dat %>% dplyr::left_join(dat_region_nuc, by = "Region")
    return(dat)
})
