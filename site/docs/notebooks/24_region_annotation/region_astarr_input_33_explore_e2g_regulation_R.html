lst = lapply(VEC_TXT_FOLDER, function(idx){
    ### get tables
    dat_region_link  = lst_dat_region_annot_e2g_link[[idx]]
    dat_region_label = lst_dat_region_annot_e2g_label[[idx]]

    ### get region significant for more than one target
    dat = dat_region_label
    dat = dat %>% dplyr::filter(Regulated_Count > 1)
    vec_txt_region = dat$Region

    ### get the singificant link with region targeted more than one gene
    dat = dat_region_link
    dat = dat %>% 
        dplyr::select(Region, Target, Regulated) %>%
        dplyr::distinct() %>%
        dplyr::filter(Regulated) %>%
        dplyr::filter(Region %in% vec_txt_region)
    
    return(dat)
})

### prepare list
lst = lst_dat_region_merge
dat = lst[[1]]
lst = split(dat$Region, dat$Target)

### generate plot
plt = upset(
    fromList(lst), 
    order.by = "freq", 
    text.scale=1.7, 
    nsets = length(lst),
    nintersects = NA,
    mb.ratio = c(0.3, 0.7)
)
