Helper function for showing percentage

Code 
fun_inner = function(num){
    if (num > 99) {
        num = round(num, 1)
        return(num)
    } 
    if (num < 1) {
        num = round(num, 1)
        return(num)  
    }
    num = round(num)    
    return(num)
}

fun_wrapper = function(vec_num_inp){
    vec_num_out = sapply(vec_num_inp, fun_inner)
    return(vec_num_out)  
}
